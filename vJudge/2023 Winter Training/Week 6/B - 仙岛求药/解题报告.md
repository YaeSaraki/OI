char grid[23][23];
int flag = 0;
int minStep = INT_MAX;
void DetermineMinStep(int m, int n, int step);

int main() {
  int M, N, m, n;
  cin >> M >> N;
  for (int i = 0; i < M; ++i) {
    for (int j = 0; j < N; ++j) {
      cin >> grid[i][j];
      if (grid[i][j] == '@') m = i, n = j;
    }
  }
  int step = 0;
  grid[m][n] = 'x';
  DetermineMinStep(m, n, step);
  cout << (flag ? minStep : -1) << endl;
  return 0;
}
void DetermineMinStep(int m, int n, int step) {
  int mm = m + 1, nn = n;
  switch (grid[mm][nn]) {
    case '.':
      grid[mm][nn] = 'x';
      DetermineMinStep(mm, nn, ++step);
      break;
    case '*':
      ++flag;
      minStep = (minStep < step) ? minStep : step;
  }
  mm = m - 1, nn = n;
  if (mm + 1) switch (grid[mm][nn]) {
      case '.':
        grid[mm][nn] = 'x';
        DetermineMinStep(mm, nn, ++step);
        break;
      case '*':
        ++flag;
        minStep = (minStep < step) ? minStep : step;
    }
  mm = m, nn = n + 1;
  switch (grid[mm][nn]) {
    case '.':
      grid[mm][nn] = 'x';
      DetermineMinStep(mm, nn, ++step);
      break;
    case '*':
      ++flag;
      minStep = (minStep < step) ? minStep : step;
  }
  mm = m, nn = n - 1;
  if (nn + 1) switch (grid[mm][nn]) {
      case '.':
        grid[mm][nn] = 'x';
        DetermineMinStep(mm, nn, ++step);
        break;
      case '*':
        ++flag;
        minStep = (minStep < step) ? minStep : step;
    }
}
