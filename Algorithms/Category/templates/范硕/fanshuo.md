```CPP
//排序结构体 
typedef struct
{
	int x;			//报名号 
	int y;			//笔试成绩 
}emxs;
int cmp(const void *a,const void *b)	//排序 
{
	emxs *p=(emxs*)a;
	emxs *q=(emxs*)b;
	if((p->y)==(q->y))
	return (p->x)-(q->x);
	if((p->y)!=(q->y))
	return (q->y)-(p->y);
}
```

### 快速幂

```CPP
//快速幂,求a^b的最后三位数		注意开long long 
//快速幂原理:底数平方,指数减半
//还运用到取模运算规则		例:(a*b)%p=(a%p*b%p)%p
typedef long long ll;
ll p;			//取模 
ll pow(ll base,ll pow)
{
	ll s=1;
	while(pow>0)
	{
		if(pow%2==1)	//指数奇数 
		{
			s = s * base % p;
		}
		pow /= 2;		//不论奇数还是偶数,除2结果相同 
		base =(base * base) % p;	//底数平方 
	}
	return s;
 }
```
